% ====================================================
% Custom macros to describe of algorithms
% ====================================================

% Macro to build input or output block.
%
% Parameters
% ----------
% #1 : The content of input or output.
%
% Returns
% -------
% \Algioitem : The input or output block.
\newcommand{\Algioitem}[1]{\State \hspace{2.5pt} #1}

% Macro to describe the algorithm
%
% Parameters
% ----------
% #1 : The caption of the algorithm.
% #2 : The label of the algorithm. Must be format `alg:myAlgLabel`
% #3 : The input.
% #4 : The output.
% #5 : The description of algorithm.
%
% Returns
% -------
% \Alg : The algorithm structure with input and output.
\newcommand{\Alg}[5]{
\begin{algorithm}
    \caption{#1}\label{#2}
    \begin{algorithmic}
        \Require
            #3
        \Ensure
            \hspace{2.5pt} #4
    \end{algorithmic}
    \noindent\rule{\textwidth}{0.6pt}
    \begin{algorithmic}[1]
        #5
    \end{algorithmic}
\end{algorithm}
}
